### 4.3 聚合引擎 `src/lib/aggregationEngine.ts`

纯函数，无副作用，可独立测试：

```typescript
import type { Aggregation } from '@/types/dashboard';

interface AggregateParams {
  data: Record<string, any>[];     // 原始数据
  groupByField: string;            // 分组维度
  valueFields: string[];           // 数值字段列表
  aggregation: Aggregation;        // 聚合方式
}

export function aggregate({
  data, groupByField, valueFields, aggregation
}: AggregateParams): Record<string, any>[] {
  if (data.length === 0) return [];

  // none = 不聚合，直接透传指定字段
  if (aggregation === 'none') {
    return data.map((row) => {
      const result: Record<string, any> = { [groupByField]: row[groupByField] };
      valueFields.forEach((f) => { result[f] = row[f]; });
      return result;
    });
  }

  // 按 groupByField 分组
  const groups = new Map<string, Record<string, any>[]>();
  for (const row of data) {
    const key = String(row[groupByField]);
    if (!groups.has(key)) groups.set(key, []);
    groups.get(key)!.push(row);
  }

  // 对每组执行聚合
  const result: Record<string, any>[] = [];
  for (const [key, rows] of groups) {
    const entry: Record<string, any> = { [groupByField]: key };
    for (const field of valueFields) {
      const values = rows.map((r) => Number(r[field]) || 0);
      switch (aggregation) {
        case 'sum':   entry[field] = values.reduce((a, b) => a + b, 0); break;
        case 'avg':   entry[field] = values.reduce((a, b) => a + b, 0) / values.length; break;
        case 'count': entry[field] = values.length; break;
        case 'max':   entry[field] = Math.max(...values); break;
      }
    }
    result.push(entry);
  }
  return result;
}
```

      return result;

### 4.6 看板状态管理 `src/hooks/useDashboardStore.ts`

```typescript
import { useState, useCallback } from 'react';
import type { CardConfig, DashboardCard, DashboardLayout, GridPosition } from '@/types/dashboard';

const STORAGE_KEY = 'dashboard-v2';
const MAX_CARDS = 20;

// 默认 3 个预设卡片
export const DEFAULT_CARDS: DashboardCard[] = [
  {
    config: { id: 'default-1', title: '产线产量完成情况', dataSourceId: 'line-production',
              chartType: 'bar', groupByField: 'lineName', valueFields: ['planned', 'actual'], aggregation: 'none' },
    grid: { x: 0, y: 0, w: 6, h: 4 },
  },
  {
    config: { id: 'default-2', title: '近7天不良率趋势', dataSourceId: 'weekly-defects',
              chartType: 'line', groupByField: 'date', valueFields: ['defectRate'], aggregation: 'none' },
    grid: { x: 6, y: 0, w: 6, h: 4 },
  },
  {
    config: { id: 'default-3', title: '不良类型分布', dataSourceId: 'quality',
              chartType: 'pie', groupByField: 'defectType', valueFields: ['defectCount'], aggregation: 'sum' },
    grid: { x: 0, y: 4, w: 6, h: 4 },
  },
];

/** 从 localStorage 加载，失败则返回默认 */
export function loadDashboard(): DashboardCard[] {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    if (!raw) return DEFAULT_CARDS;
    const parsed = JSON.parse(raw) as DashboardLayout;
    if (parsed.version !== 2 || !Array.isArray(parsed.cards)) return DEFAULT_CARDS;
    return parsed.cards;
  } catch {
    return DEFAULT_CARDS;
  }
}

/** 保存到 localStorage */
export function saveDashboard(cards: DashboardCard[]): void {
  try {
    localStorage.setItem(STORAGE_KEY, JSON.stringify({ version: 2, cards }));
  } catch {
    // localStorage 满时静默失败，由调用方处理提示
  }
}

export function useDashboardStore() {
  const [cards, setCards] = useState<DashboardCard[]>(() => loadDashboard());
  const [snapshot, setSnapshot] = useState<DashboardCard[]>(() => loadDashboard());
  const [isEditing, setIsEditing] = useState(false);

  const startEditing = useCallback(() => { setSnapshot([...cards]); setIsEditing(true); }, [cards]);

  const save = useCallback(() => { saveDashboard(cards); setSnapshot(cards); setIsEditing(false); }, [cards]);

  const cancel = useCallback(() => { setCards(snapshot); setIsEditing(false); }, [snapshot]);

  const addCard = useCallback((config: CardConfig) => {
    setCards((prev) => {
      if (prev.length >= MAX_CARDS) return prev; // 卡片上限
      const shifted = prev.map(c => ({ ...c, grid: { ...c.grid, y: c.grid.y + 4 } }));
      return [{ config, grid: { x: 0, y: 0, w: 6, h: 4 } }, ...shifted];
    });
  }, []);

  const updateCard = useCallback((cardId: string, updates: Partial<CardConfig>) => {
    setCards((prev) => prev.map((c) => c.config.id === cardId ? { ...c, config: { ...c.config, ...updates } } : c));
  }, []);

  const deleteCard = useCallback((cardId: string) => {
    setCards((prev) => prev.filter((c) => c.config.id !== cardId));
  }, []);

  const updateLayout = useCallback((positions: Array<{ id: string; grid: GridPosition }>) => {
    setCards((prev) => prev.map((c) => {
      const pos = positions.find((p) => p.id === c.config.id);
      return pos ? { ...c, grid: pos.grid } : c;
    }));
  }, []);

  const resetToDefault = useCallback(() => {
    setCards(DEFAULT_CARDS);
    saveDashboard(DEFAULT_CARDS);
  }, []);

  return { cards, isEditing, startEditing, save, cancel, addCard, updateCard, deleteCard, updateLayout, resetToDefault };
}
```

### 4.8 卡片渲染器 `src/components/DashboardCardRenderer.tsx`

每个卡片独立获取数据并渲染图表：

1. 调用 `getDataSourceData(config.dataSourceId)` 获取原始数据
2. 调用 `aggregate()` 按配置聚合
3. 按 `config.chartType` 渲染 Recharts 图表
4. 卡片头部：标题 + 图表类型切换按钮组

**图表渲染规则：**

## 七、风险与注意事项

| 风险 | 影响 | 应对 |
|------|------|------|
| gridstack.js 与 React 状态同步 | 拖拽后 React state 和 DOM 不一致 | 在 `grid.on('change')` 中同步，用 `batchUpdate` 批量操作 |
| gridstack CSS 导入 | Vite 构建时可能找不到 CSS | 使用 `import 'gridstack/dist/gridstack.min.css'` 绝对路径 |
| Recharts ResponsiveContainer 高度 | 在 gridstack 绝对定位容器中百分比高度不生效 | 固定高度 280px |
| localStorage 容量 | 超出 5MB 限制 | save 时 try-catch，失败提示用户 |
| 前后端数据源字段定义不一致 | 前端字段与后端返回字段不匹配 | 前后端各维护一份 metaMap，上线前对齐 |
